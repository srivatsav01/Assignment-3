\section*{GATE 1999(MA) Question 1.29}
A box contains two coins,one of which is fair and the other is two headed.One coin is choosen at random and tossed twice.If two heads appear,then  the probability that the choosen coin is two headed is?

\section*{Solution}
   
     Let A be the event that choosen coin is two headed.\\
     Let B be the event that choosen coin is fair.\\
     Let X be the random variable representing no of heads.\\
     
     if event A occurs\\
     \Pr\brak{X=2}=\Pr\brak{A}\\
     
     if event B occurs Probability of getting a head is $\frac{1}{2}$ \\
     
     $\therefore\Pr\brak{X=2}=\Pr\brak{B}\times\brak{\frac{1}{2}}^{2}$\\
     
     As events A and B are exclusive\
     
     
  $$\therefore \Pr\brak{X=2} =\brak{\Pr\brak{B}\times\brak{\frac{1}{2}}^{2}}+\brak{\Pr\brak{A}}$$
 
    \begin{align}
    \centering
  \tag{1}
  \Pr\brak{X=2}&=\brak{\frac{1}{2}\times\brak{\fr\frac{1}{2}}^{2}}+\brak{\frac{1}{2}}\label{eq1}\\
  \tag{2}
  \Pr\brak{X=2}&=\frac{5}{8}\label{eq2}\\
  \tag{3}
  \Pr\brak{A |\brak{X=2}}&=\frac{\Pr\brak{A,\brak{X=2}}}{\Pr\brak{X=2}}\label{eq3}\\
  \tag{4}
  \Pr\brak{A |\brak{X=2}}&=\frac{\brak{\frac{1}{2}}}{\brak{\frac{5}{8}}}\label{eq4}\\
  \tag{5}
  \Pr\brak{A|\brak{X=2}}&=\frac{4}{5}\label{eq5}
   \end{align}
